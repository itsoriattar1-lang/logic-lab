function renderContent(data) { const out = document.getElementById('output'); out.innerHTML = ""; // RULE 1: STRICT SKIP OF PAGE 1 const realData = data.slice(1); // Global sentence pool for high-quality quiz distractors const allSentences = data.flatMap(p => p.text.split(/[.!?]+/).map(s => s.trim()).filter(s => s.length > 50)); realData.forEach((page, index) => { // CLEANING & EXTRACTION const rawSentences = page.text.split(/[.!?]+/).map(s => s.trim()).filter(s => s.length > 30); // Helper to build clean, professional headers const getContextFragment = (sentence, wordCount) => { if (!sentence) return "this specialized section"; return sentence.split(' ').slice(0, wordCount).join(' ').replace(/[,.;:]$/, ""); }; const primaryThesis = getContextFragment(rawSentences[0], 10); const supportingDetail = rawSentences[1] ? getContextFragment(rawSentences[1], 12) : "additional thematic evidence"; const logicChain = rawSentences.find(s => s.match(/because|since|therefore|resulted|consequently|witnessed/i)) || rawSentences[0]; const card = document.createElement('div'); card.className = "study-card p-12 mb-12 shadow-sm border border-slate-100 bg-white animate-in fade-in slide-in-from-bottom-4"; let contentHTML = ""; // MODE 1: DEEP SUMMARY & SECTION CONTEXT if (currentLevel === "l1") { contentHTML = `

Thematic Overview

This section primarily examines **${primaryThesis}...** while offering a detailed look into **${supportingDetail}...**

Academic Context

By focusing on ${primaryThesis}, this page establishes a critical link between the core narrative and the supporting facts found in ${supportingDetail}. This provides the reader with a comprehensive understanding of the section's objectives.

`; } // MODE 2: CAUSAL LOGIC BREAKDOWN else if (currentLevel === "l2") { contentHTML = `

Logic & Development

Analyzing the causal relationships within this page:

${logicChain}.

`; } // MODE 3: MASTERY & EVIDENCE VERIFICATION else { const correctClaim = rawSentences[0] || "Confirmed textual evidence."; const distractor = allSentences.filter(s => !s.includes(primaryThesis))[Math.floor(Math.random() * 10)] || "General background information."; const options = [{ text: correctClaim, isCorrect: true }, { text: distractor, isCorrect: false }].sort(() => Math.random() - 0.5); contentHTML = `

Evidence-Based Challenge

Topic: ${primaryThesis}

Which of these specific assertions is directly supported by the text on Page ${page.num}?

${options.map(opt => `
?

${opt.text}

`).join('')}
`; } card.innerHTML = `
PAGE ${page.num}
Verified Analysis Thread
${contentHTML}`; out.appendChild(card); }); }
