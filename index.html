function renderContent(data) { const out = document.getElementById('output'); out.innerHTML = ""; data.forEach((page, index) => { const text = page.text; // 1. REFINED EXTRACTION: Finds sentences that aren't just empty space const sentences = text.split(/[.!?]+/).map(s => s.trim()).filter(s => s.length > 25); // 2. SMART KEYWORD SCANNER: Specifically hunts for nouns/proper names (5+ letters) const noise = ['the', 'this', 'that', 'with', 'from', 'their', 'which', 'about', 'would', 'there', 'could']; const words = text.match(/\b[A-Z]?[a-z]{5,}\b/g) || []; const keywords = [...new Set(words.filter(w => !noise.includes(w.toLowerCase())))].slice(0, 6); // 3. DYNAMIC INTRO: No more "this section" placeholders // It now grabs the first 6 words of the page as the "Main Theme" const mainTheme = sentences[0] ? sentences[0].split(' ').slice(0, 6).join(' ') : "General Content"; const subTheme = sentences[1] ? sentences[1].split(' ').slice(0, 6).join(' ') : "Supporting Details"; const descriptiveIntro = `This page analyzes **${mainTheme}...** while primarily focusing on **${subTheme}...**`; const card = document.createElement('div'); card.className = "study-card p-10 mb-8 shadow-sm animate-in fade-in slide-in-from-bottom-4 duration-500"; let inner = ""; if (currentLevel === "l1") { inner = `

${descriptiveIntro}

Detected Keywords

${keywords.map(w => `#${w.toUpperCase()}`).join('')}
`; } else if (currentLevel === "l2") { const coreLogic = sentences.find(s => s.match(/is|was|because|due to|result/i)) || sentences[0]; inner = `

Logic Breakdown

${descriptiveIntro}

${coreLogic}.

`; } else { const choices = [ { t: sentences[0] || "Primary source evidence", c: true }, { t: "General summary of unrelated document meta-data.", c: false } ].sort(() => Math.random() - 0.5); inner = `

${descriptiveIntro}

Mastery Challenge:

${choices.map(o => `
${o.t}
`).join('')}
VERIFIED KEYWORDS: ${keywords.join(' • ')}
`; } card.innerHTML = `
DATA STREAM: PAGE ${page.num}
${inner}`; out.appendChild(card); }); }
