function renderContent(data) { const out = document.getElementById('output'); out.innerHTML = ""; // RULE 1: NEVER ANALYZE PAGE 1 (Title/Index) const realData = data.slice(1); const allSentences = data.flatMap(p => p.text.split(/[.!?]+/).map(s => s.trim()).filter(s => s.length > 50)); realData.forEach((page) => { const rawSentences = page.text.split(/[.!?]+/).map(s => s.trim()).filter(s => s.length > 35); // --- POWER WORD SCANNER (Fixes the "This section" error) --- const getHighValueContext = (sentenceList) => { if (!sentenceList.length) return "specialized thematic data"; // Finds the longest sentence in the first 3 lines (usually the most descriptive) const target = sentenceList.slice(0, 3).reduce((a, b) => a.length > b.length ? a : b); // Extracts words that aren't common (the, is, and) to find the actual topic const stopWords = ['this', 'page', 'section', 'talks', 'about', 'the', 'and', 'with', 'from']; const words = target.split(' ').filter(w => !stopWords.includes(w.toLowerCase())); // Returns a 10-word chunk starting from the first "real" word return words.slice(0, 10).join(' ').replace(/[,.;:]$/, ""); }; const primaryThesis = getHighValueContext(rawSentences); const secondaryContext = rawSentences[2] ? getHighValueContext(rawSentences.slice(2)) : "contextual evidence"; const logicChain = rawSentences.find(s => s.match(/because|since|therefore|resulted|consequently|witnessed/i)) || rawSentences[0]; const card = document.createElement('div'); card.className = "study-card p-12 mb-12 shadow-sm border border-slate-100 bg-white animate-in fade-in slide-in-from-bottom-4"; let contentHTML = ""; // MODE 1: DEEP SUMMARY if (currentLevel === "l1") { contentHTML = `

Core Thematic Analysis

This page centers on the analysis of ${primaryThesis}...

Supporting Framework

The text builds its primary argument by referencing ${secondaryContext}, creating a link between the abstract themes and the specific evidence presented on this page.

`; } // MODE 2: CAUSAL LOGIC else if (currentLevel === "l2") { contentHTML = `

Causal Logic Flow

${logicChain}.

`; } // MODE 3: MASTERY CHALLENGE else { const correctClaim = rawSentences[0] || "Verified claim."; const distractor = allSentences.filter(s => !s.includes(primaryThesis.split(' ')[0]))[Math.floor(Math.random() * 10)] || "General annotation."; const options = [{ text: correctClaim, isCorrect: true }, { text: distractor, isCorrect: false }].sort(() => Math.random() - 0.5); contentHTML = `

Verification: ${primaryThesis}

${options.map(opt => `
?

${opt.text}

`).join('')}
`; } card.innerHTML = `
PAGE ${page.num}
ENGINE ANALYSIS ACTIVE
${contentHTML}`; out.appendChild(card); }); }
