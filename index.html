function renderContent(data) { const out = document.getElementById('output'); out.innerHTML = ""; data.forEach((page, index) => { const text = page.text; // 1. IMPROVED DATA EXTRACTION const sentences = text.split('. ').filter(s => s.length > 20); const dates = text.match(/\b(17|18|19|20)\d{2}\b/g) || []; // 2. KEYWORD GENERATOR (Filter out common words like 'the', 'is', 'and') const commonWords = ['the', 'this', 'that', 'with', 'from', 'they', 'their', 'which', 'about']; const words = text.toLowerCase().match(/\b[a-z]{5,}\b/g) || []; const keywords = [...new Set(words.filter(w => !commonWords.includes(w)))].slice(0, 5); // 3. SENTENCE BUILDER: "This page talks about..." // Grabs the first few words of the first two meaningful sentences const topicA = sentences[0] ? sentences[0].split(' ').slice(0, 5).join(' ') : "this section"; const topicB = sentences[1] ? sentences[1].split(' ').slice(0, 5).join(' ') : "related details"; const descriptiveIntro = `This page talks about **${topicA}** and covers **${topicB}**.`; const card = document.createElement('div'); card.className = "study-card p-10 shadow-sm mb-8 animate-in fade-in slide-in-from-bottom-4 duration-500"; let html = ""; if (currentLevel === "l1") { html = `

${descriptiveIntro}

Key Terminology

${keywords.map(word => `#${word}`).join('')}
`; } else if (currentLevel === "l2") { const conclusion = sentences.find(s => s.match(/therefore|because|result|since/i)) || sentences[0]; html = `

Logic Breakdown

${descriptiveIntro}

Core Logic: ${conclusion}

`; } else { const choices = [ { t: sentences[0], c: true }, { t: "This page focuses on minor technical annotations.", c: false } ].sort(() => Math.random() - 0.5); html = `

${descriptiveIntro}

Mastery Challenge:

${choices.map(o => `
${o.t}
`).join('')}
KEYWORDS: ${keywords.join(', ')}
`; } card.innerHTML = `
PAGE ${page.num}
${dates.length > 0 ? `TIMELINE: ${dates[0]}` : ''}
${html}`; out.appendChild(card); }); }
